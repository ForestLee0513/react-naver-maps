import { useState, useRef } from 'react';
import { MapDiv, NaverMap, useNavermaps } from 'react-naver-maps';

# 지도 옵션 조정하기

<MapDiv style={{
  width: '100%',
  height: '600px',
}}>
  {() => {
    const navermaps = useNavermaps();
    const [mapTypeId, setMapTypeId] = useState(navermaps.MapTypeId.NORMAL);
    const center = useRef(new navermaps.LatLng(37.3595704, 127.105399)).current;
    const buttons = [
      { 
        typeId: navermaps.MapTypeId.NORMAL,
        text: '일반지도' 
      },
      { 
        typeId: navermaps.MapTypeId.TERRAIN,
        text: '지형도' 
      },
      { 
        typeId: navermaps.MapTypeId.SATELLITE,
        text: '위성지도' 
      },
      { 
        typeId: navermaps.MapTypeId.HYBRID,
        text: '겹쳐보기' 
      },
    ]

    return (
      <>
        {buttons.map(btn => {
          return <button
            style={{
              backgroundColor: btn.typeId === mapTypeId ? '#2780E3' : 'white',
              color: btn.typeId === mapTypeId ? 'white' : 'black',
            }}
            onClick={() => {
              setMapTypeId(btn.typeId)
            }}>{btn.text}</button>
        })}
        <NaverMap 
          zoom={15}
          mapTypeId={mapTypeId}
          center={center}
        />
      </>
    );
  }}
</MapDiv>
